function schedulePosts(posts, platforms) {
  const phantomEngine = new PhantomCore();
  phantomEngine.authenticate(API_KEYS);
  
  posts.forEach((post, index) => {
    const delay = index * INTERVAL;
    setTimeout(() => {
      platforms.forEach(platform => {
        phantomEngine.postToPlatform(platform, post);
      });
    }, delay);
  });
}
